下面是 **Echive 的 Prisma Schema 初稿**，按你当前 MVP 设计，可直接作为起点。

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum IdeaType {
  IDEA
  TASK
  CONTENT
  PROJECT
}

enum IdeaStatus {
  CAPTURED
  CLARIFIED
  PLANNED
  ARCHIVED
}

enum ProjectStatus {
  ACTIVE
  PAUSED
  DONE
  ARCHIVED
}

enum TaskStatus {
  TODO
  DOING
  BLOCKED
  DONE
}

enum TaskPriority {
  LOW
  MEDIUM
  HIGH
}

enum MaterialType {
  NOTE
  LINK
  QUOTE
  IMAGE
  VOICE
  CHAT
}

enum MaterialStatus {
  RAW
  SORTED
  USABLE
  USED
}

enum DraftFormat {
  BLOG
  WEIBO
  VLOG
  SCRIPT
}

enum DraftStatus {
  DRAFT
  REVIEWING
  READY
  PUBLISHED
}

model User {
  id        String    @id @default(cuid())
  email     String?   @unique
  name      String?
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt

  ideas     Idea[]
  projects  Project[]
  tasks     Task[]
  materials Material[]
  drafts    Draft[]
}

model Project {
  id          String        @id @default(cuid())
  userId      String
  title       String
  description String?       @db.Text
  goal        String?       @db.Text
  status      ProjectStatus @default(ACTIVE)
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt

  user        User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  ideas       Idea[]
  tasks       Task[]
  materials   Material[]
  drafts      Draft[]
  tags        ProjectTag[]

  @@index([userId])
  @@index([status])
}

model Idea {
  id          String      @id @default(cuid())
  userId      String
  projectId   String?
  title       String
  rawInput    String      @db.Text
  summary     String?     @db.Text
  type        IdeaType    @default(IDEA)
  status      IdeaStatus  @default(CAPTURED)
  nextAction  String?     @db.Text
  createdAt   DateTime    @default(now())
  updatedAt   DateTime    @updatedAt

  user        User        @relation(fields: [userId], references: [id], onDelete: Cascade)
  project     Project?    @relation(fields: [projectId], references: [id], onDelete: SetNull)
  tasks       Task[]
  materials   Material[]
  drafts      Draft[]
  tags        IdeaTag[]

  @@index([userId])
  @@index([projectId])
  @@index([status])
  @@index([type])
}

model Task {
  id          String       @id @default(cuid())
  userId      String
  projectId   String?
  ideaId      String?
  title       String
  description String?      @db.Text
  status      TaskStatus   @default(TODO)
  priority    TaskPriority @default(MEDIUM)
  dueDate     DateTime?
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt

  user        User         @relation(fields: [userId], references: [id], onDelete: Cascade)
  project     Project?     @relation(fields: [projectId], references: [id], onDelete: SetNull)
  idea        Idea?        @relation(fields: [ideaId], references: [id], onDelete: SetNull)
  tags        TaskTag[]

  @@index([userId])
  @@index([projectId])
  @@index([ideaId])
  @@index([status])
  @@index([priority])
  @@index([dueDate])
}

model Material {
  id          String         @id @default(cuid())
  userId      String
  projectId   String?
  ideaId      String?
  title       String
  type        MaterialType
  content     String         @db.Text
  source      String?
  status      MaterialStatus @default(RAW)
  createdAt   DateTime       @default(now())
  updatedAt   DateTime       @updatedAt

  user        User           @relation(fields: [userId], references: [id], onDelete: Cascade)
  project     Project?       @relation(fields: [projectId], references: [id], onDelete: SetNull)
  idea        Idea?          @relation(fields: [ideaId], references: [id], onDelete: SetNull)
  drafts      DraftMaterial[]
  tags        MaterialTag[]

  @@index([userId])
  @@index([projectId])
  @@index([ideaId])
  @@index([status])
  @@index([type])
}

model Draft {
  id          String        @id @default(cuid())
  userId      String
  projectId   String?
  ideaId      String?
  title       String
  format      DraftFormat
  outline     String?       @db.Text
  content     String        @db.Text
  status      DraftStatus   @default(DRAFT)
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt

  user        User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  project     Project?      @relation(fields: [projectId], references: [id], onDelete: SetNull)
  idea        Idea?         @relation(fields: [ideaId], references: [id], onDelete: SetNull)
  materials   DraftMaterial[]
  tags        DraftTag[]

  @@index([userId])
  @@index([projectId])
  @@index([ideaId])
  @@index([status])
  @@index([format])
}

model DraftMaterial {
  draftId     String
  materialId  String

  draft       Draft    @relation(fields: [draftId], references: [id], onDelete: Cascade)
  material    Material @relation(fields: [materialId], references: [id], onDelete: Cascade)

  @@id([draftId, materialId])
  @@index([materialId])
}

model Tag {
  id          String        @id @default(cuid())
  name        String        @unique
  createdAt   DateTime      @default(now())

  ideaTags     IdeaTag[]
  projectTags  ProjectTag[]
  taskTags     TaskTag[]
  materialTags MaterialTag[]
  draftTags    DraftTag[]
}

model IdeaTag {
  ideaId String
  tagId  String

  idea   Idea @relation(fields: [ideaId], references: [id], onDelete: Cascade)
  tag    Tag  @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([ideaId, tagId])
  @@index([tagId])
}

model ProjectTag {
  projectId String
  tagId     String

  project   Project @relation(fields: [projectId], references: [id], onDelete: Cascade)
  tag       Tag     @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([projectId, tagId])
  @@index([tagId])
}

model TaskTag {
  taskId String
  tagId  String

  task   Task @relation(fields: [taskId], references: [id], onDelete: Cascade)
  tag    Tag  @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([taskId, tagId])
  @@index([tagId])
}

model MaterialTag {
  materialId String
  tagId      String

  material   Material @relation(fields: [materialId], references: [id], onDelete: Cascade)
  tag        Tag      @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([materialId, tagId])
  @@index([tagId])
}

model DraftTag {
  draftId String
  tagId   String

  draft   Draft @relation(fields: [draftId], references: [id], onDelete: Cascade)
  tag     Tag   @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([draftId, tagId])
  @@index([tagId])
}
```

---

## 这版 schema 的特点
1. **覆盖 MVP 核心对象**
   - User
   - Idea
   - Project
   - Task
   - Material
   - Draft

2. **支持标签系统**
   - 统一 `Tag`
   - 各实体通过中间表关联

3. **支持素材与草稿多对多**
   - 一个 Draft 可引用多条 Material
   - 一条 Material 也可复用在多个 Draft

4. **支持后续 AI 扩展**
   - `rawInput`
   - `summary`
   - `nextAction`
   - `outline`
   这些都为后续 AI 自动生成留了位置
